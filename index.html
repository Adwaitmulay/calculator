#include <iostream>
using namespace std;

void showMenu() {
    cout << "\n=== Simple Calculator ===" << endl;
    cout << "1. Addition       (+)" << endl;
    cout << "2. Subtraction    (-)" << endl;
    cout << "3. Multiplication (*)" << endl;
    cout << "4. Division       (/)" << endl;
    cout << "5. Modulus        (%)" << endl;
    cout << "0. Exit" << endl;
    cout << "Enter choice: ";
}

double calculate(double a, double b,
                 int choice, bool &valid) {
    valid = true;
    switch (choice) {
        case 1: return a + b;
        case 2: return a - b;
        case 3: return a * b;
        case 4:
            if (b == 0) {
                cout << "Error: Division by zero."
                     << endl;
                valid = false; return 0;
            }
            return a / b;
        case 5:
            if (b == 0) {
                cout << "Error: Modulus by zero."
                     << endl;
                valid = false; return 0;
            }
            return (int)a % (int)b;
        default:
            cout << "Error: Invalid operation." << endl;
            valid = false; return 0;
    }
}

int main() {
    int choice;
    double num1, num2;
    bool valid;

    cout << "Welcome to Simple Calculator!" << endl;

    while (true) {
        showMenu();
        cin >> choice;
        if (choice == 0) {
            cout << "Exiting. Goodbye!" << endl;
            break;
        }
        cout << "Enter first number: ";
        cin >> num1;
        cout << "Enter second number: ";
        cin >> num2;
        double result = calculate(num1, num2,
                                  choice, valid);
        if (valid)
            cout << "Result: " << result << endl;
    }
    return 0;
}
Code breakdown
⚙️ Modular design
Logic split into showMenu() and calculate() — single responsibility per function, reducing duplication.
showMenu() · calculate() · main()
🔀 Switch-case control flow
All 5 operations handled via a clean switch — readable, extendable, efficient for discrete inputs.
switch(choice) { case 1..5 }
🛡️ Edge case handling
Division and modulus by zero caught before execution — program recovers gracefully without crashing.
if (b == 0) → Error + valid=false
🔁 Persistent loop
while(true) keeps the calculator running — clean exit on input 0, no memory leaks or dangling state.
while(true) · break on exit
